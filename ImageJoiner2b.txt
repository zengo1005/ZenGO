import React, { useState, useRef, useEffect, useCallback } from 'react';
import { UploadCloud, Image as ImageIcon, Trash2, Download, Settings, FileImage, AlertCircle } from 'lucide-react';

export default function App() {
  const [images, setImages] = useState([]);
  const [columns, setColumns] = useState(3);
  const [totalWidth, setTotalWidth] = useState(1200);
  const [downloadFileName, setDownloadFileName] = useState('combined_images');
  const [downloadFormat, setDownloadFormat] = useState('jpeg');
  const [jpegQuality, setJpegQuality] = useState(0.9);
  const [isDragging, setIsDragging] = useState(false);
  const [isProcessing, setIsProcessing] = useState(false);
  const [previewUrl, setPreviewUrl] = useState(null);
  const [error, setError] = useState(null);

  const fileInputRef = useRef(null);

  // ファイルを処理し、Stateを更新する
  const handleFiles = useCallback(async (newFiles) => {
    setError(null);
    
    // 画像ファイルのみを抽出
    const validNewFiles = Array.from(newFiles).filter(f => f.type.startsWith('image/'));
    if (validNewFiles.length === 0) return;

    setIsProcessing(true);

    setImages(prevImages => {
      const existingFiles = prevImages.map(img => img.file);
      let allFiles = [...existingFiles, ...validNewFiles];

      // ファイル名称で自然順ソート (例: img2.jpg が img10.jpg より前に来るように)
      allFiles.sort((a, b) => a.name.localeCompare(b.name, undefined, { numeric: true, sensitivity: 'base' }));

      // ファイル名で重複排除
      const uniqueFilesMap = new Map();
      allFiles.forEach(f => uniqueFilesMap.set(f.name, f));
      allFiles = Array.from(uniqueFilesMap.values());

      // 上限255個の制限
      if (allFiles.length > 255) {
        setError('選択できる画像の数は上限255個です。255個に制限されました。');
        allFiles = allFiles.slice(0, 255);
      }

      // 非同期で画像を読み込み (既存のオブジェクトURLは解放)
      prevImages.forEach(img => URL.revokeObjectURL(img.src));

      const loadImagesAsync = async () => {
        const imagePromises = allFiles.map(file => {
          return new Promise((resolve) => {
            const url = URL.createObjectURL(file);
            const img = new Image();
            img.onload = () => resolve({
              file,
              name: file.name,
              src: url,
              width: img.width,
              height: img.height,
              element: img
            });
            img.onerror = () => {
              URL.revokeObjectURL(url);
              resolve(null); // エラー時はnull
            };
            img.src = url;
          });
        });

        const loadedImages = (await Promise.all(imagePromises)).filter(img => img !== null);
        setImages(loadedImages);
        setIsProcessing(false);
      };

      loadImagesAsync();
      
      // 更新中のテンポラリ配列として空を返す（実際の更新は loadImagesAsync 内で行う）
      // ※ここでは連続更新を防ぐため一度配列をクリアするような挙動になります
      return []; 
    });
  }, []);

  // 画像結合ロジック
  const generateImage = useCallback(() => {
    if (images.length === 0) {
      setPreviewUrl(null);
      setIsProcessing(false);
      return;
    }

    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');

    // 指定された列数で画像をチャンク（行）に分割
    const chunks = [];
    for (let i = 0; i < images.length; i += columns) {
      chunks.push(images.slice(i, i + columns));
    }

    let currentY = 0;
    let previousRowHeight = 0;
    
    // 各行と画像のサイズを計算
    const rowsData = chunks.map((chunk, index) => {
      const isLastChunk = index === chunks.length - 1;
      let rowHeight = 0;
      
      // 行内の画像のアスペクト比の合計
      const totalAspectRatio = chunk.reduce((sum, img) => sum + (img.width / img.height), 0);

      if (isLastChunk && chunk.length < columns && index > 0) {
        // 最後の行で画像数が足りない場合、直前の行の高さを維持して左詰めに配置する
        rowHeight = previousRowHeight;
      } else {
        // 指定された全体幅にぴったり合うように行の高さを計算
        rowHeight = totalWidth / totalAspectRatio;
      }
      
      previousRowHeight = rowHeight;

      const rowData = {
        y: currentY,
        height: rowHeight,
        images: chunk.map(img => {
          const aspectRatio = img.width / img.height;
          const width = rowHeight * aspectRatio;
          return { ...img, renderWidth: width, renderHeight: rowHeight };
        })
      };

      currentY += rowHeight;
      return rowData;
    });

    const totalHeight = currentY;

    // キャンバスサイズの設定
    canvas.width = totalWidth;
    canvas.height = totalHeight;

    // 背景を白で塗りつぶす（最終行の余白部分などのため）
    ctx.fillStyle = '#ffffff';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // 画像の描画
    rowsData.forEach(row => {
      let currentX = 0;
      row.images.forEach(imgData => {
        ctx.drawImage(
          imgData.element,
          currentX,
          row.y,
          imgData.renderWidth,
          imgData.renderHeight
        );
        currentX += imgData.renderWidth;
      });
    });

    const mimeType = downloadFormat === 'jpeg' ? 'image/jpeg' : 'image/png';
    const quality = downloadFormat === 'jpeg' ? jpegQuality : undefined;

    // キャンバスをBlobに変換してプレビュー表示（メモリ対策）
    canvas.toBlob((blob) => {
      if (blob) {
        const url = URL.createObjectURL(blob);
        setPreviewUrl(prev => {
          if (prev) URL.revokeObjectURL(prev);
          return url;
        });
      } else {
        setError('プレビューの生成に失敗しました。画像サイズが大きすぎる可能性があります。');
      }
      setIsProcessing(false);
    }, mimeType, quality);

  }, [images, columns, totalWidth, downloadFormat, jpegQuality]);

  // 設定や画像リストが変更されたら画像を再生成
  useEffect(() => {
    let timer;
    if (images.length > 0) {
      setIsProcessing(true);
      // 入力の連続変更時にUIがフリーズしないようデバウンス処理
      timer = setTimeout(() => {
        generateImage();
      }, 300);
    } else {
      setPreviewUrl(null);
      setIsProcessing(false);
    }
    return () => clearTimeout(timer);
  }, [images, columns, totalWidth, generateImage]);

  // アプリケーション終了時のメモリ解放
  useEffect(() => {
    return () => {
      images.forEach(img => URL.revokeObjectURL(img.src));
      if (previewUrl) URL.revokeObjectURL(previewUrl);
    };
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

  // ドラッグ＆ドロップのイベントハンドラ
  const onDragOver = useCallback((e) => {
    e.preventDefault();
    setIsDragging(true);
  }, []);

  const onDragLeave = useCallback((e) => {
    e.preventDefault();
    if (!e.currentTarget.contains(e.relatedTarget)) {
      setIsDragging(false);
    }
  }, []);

  const onDrop = useCallback((e) => {
    e.preventDefault();
    setIsDragging(false);
    if (e.dataTransfer.files && e.dataTransfer.files.length > 0) {
      handleFiles(e.dataTransfer.files);
    }
  }, [handleFiles]);

  const removeImage = (indexToRemove) => {
    setImages(prev => {
      const newImages = [...prev];
      URL.revokeObjectURL(newImages[indexToRemove].src);
      newImages.splice(indexToRemove, 1);
      return newImages;
    });
  };

  const clearAll = () => {
    images.forEach(img => URL.revokeObjectURL(img.src));
    setImages([]);
    setPreviewUrl(null);
    setError(null);
  };

  const downloadImage = () => {
    if (!previewUrl) return;
    const a = document.createElement('a');
    a.href = previewUrl;
    const extension = downloadFormat === 'jpeg' ? 'jpg' : 'png';
    const fileName = downloadFileName.trim() || 'combined_images';
    a.download = `${fileName}.${extension}`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
  };

  return (
    <div 
      className="min-h-screen bg-slate-50 flex flex-col font-sans text-slate-800 relative"
      onDragOver={onDragOver} 
      onDragLeave={onDragLeave} 
      onDrop={onDrop}
    >
      {/* 全画面ドラッグオーバーレイ */}
      {isDragging && (
        <div className="absolute inset-0 bg-indigo-500/20 z-50 flex items-center justify-center border-4 border-indigo-500 border-dashed m-4 rounded-xl backdrop-blur-sm pointer-events-none">
          <p className="text-4xl font-extrabold text-indigo-700 bg-white/80 px-8 py-4 rounded-full shadow-lg">
            ドロップして画像を追加
          </p>
        </div>
      )}

      {/* ヘッダー */}
      <header className="bg-white shadow-sm px-6 py-4 flex items-center justify-between z-10 border-b border-slate-200">
        <div className="flex items-center gap-2 text-indigo-600">
          <ImageIcon size={28} className="stroke-2" />
          <h1 className="text-xl font-bold tracking-tight">Image Joiner</h1>
        </div>
        <p className="text-sm text-slate-500 font-medium hidden sm:block">ドラッグ＆ドロップで画像を自動レイアウト結合</p>
      </header>

      {/* メインコンテンツ */}
      <main className="flex-1 flex flex-col lg:flex-row overflow-hidden relative">
        
        {/* 左サイドバー: コントロールパネル */}
        <aside className="w-full lg:w-96 bg-white border-r border-slate-200 flex flex-col h-auto lg:h-full z-10 shadow-lg lg:shadow-none flex-shrink-0">
          <div className="p-5 flex-1 overflow-y-auto flex flex-col gap-8">
            
            {/* アップロードエリア */}
            <div 
              className="border-2 border-dashed border-slate-300 rounded-2xl p-8 flex flex-col items-center justify-center text-center cursor-pointer hover:bg-indigo-50 hover:border-indigo-400 transition-all group"
              onClick={() => fileInputRef.current?.click()}
            >
               <UploadCloud className="text-slate-400 group-hover:text-indigo-500 mb-3 transition-colors" size={42} />
               <p className="text-sm text-slate-700 font-bold mb-1">クリックまたはドラッグ＆ドロップ</p>
               <p className="text-xs text-slate-500 font-medium">最大255個まで選択可能</p>
               <input 
                 type="file" 
                 multiple 
                 accept="image/*" 
                 className="hidden" 
                 ref={fileInputRef} 
                 onChange={(e) => {
                   handleFiles(e.target.files);
                   e.target.value = ''; // 同じファイルを連続で選べるようにリセット
                 }} 
               />
            </div>

            {/* 結合設定 */}
            <div className="space-y-5 bg-slate-50 p-5 rounded-2xl border border-slate-100">
              <h2 className="text-sm font-bold text-slate-800 uppercase tracking-widest flex items-center gap-2">
                <Settings size={16} className="text-slate-500"/> 設定
              </h2>
              
              <div className="space-y-4">
                <div>
                  <label className="flex items-center justify-between text-sm font-semibold text-slate-700 mb-1.5">
                    <span>横に並べる数 (列)</span>
                    <span className="text-indigo-600 font-bold">{columns} 列</span>
                  </label>
                  <input 
                    type="range" min="1" max="10" value={columns} 
                    onChange={(e) => setColumns(parseInt(e.target.value) || 1)} 
                    className="w-full accent-indigo-600" 
                  />
                  <div className="flex justify-between text-xs text-slate-400 mt-1 px-1">
                    <span>1</span><span>5</span><span>10</span>
                  </div>
                </div>

                <div>
                  <label className="block text-sm font-semibold text-slate-700 mb-1.5">全体の横幅 (px)</label>
                  <div className="relative">
                    <input 
                      type="number" min="100" max="8000" step="100" value={totalWidth} 
                      onChange={(e) => setTotalWidth(Math.max(100, parseInt(e.target.value) || 100))} 
                      className="w-full border border-slate-300 rounded-lg pl-3 pr-10 py-2.5 text-sm font-medium focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 outline-none transition-all" 
                    />
                    <span className="absolute right-3 top-1/2 -translate-y-1/2 text-slate-400 text-sm font-medium">px</span>
                  </div>
                </div>

                <hr className="border-slate-200 my-2" />

                <div>
                  <label className="block text-sm font-semibold text-slate-700 mb-1.5">保存ファイル名</label>
                  <input 
                    type="text" value={downloadFileName} 
                    onChange={(e) => setDownloadFileName(e.target.value)} 
                    placeholder="combined_images"
                    className="w-full border border-slate-300 rounded-lg px-3 py-2 text-sm font-medium focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 outline-none transition-all"
                  />
                </div>

                <div>
                  <label className="block text-sm font-semibold text-slate-700 mb-1.5">保存形式</label>
                  <div className="flex gap-4">
                    <label className="flex items-center gap-2 text-sm font-medium cursor-pointer">
                      <input 
                        type="radio" value="jpeg" 
                        checked={downloadFormat === 'jpeg'} 
                        onChange={() => setDownloadFormat('jpeg')} 
                        className="accent-indigo-600 w-4 h-4 cursor-pointer" 
                      />
                      JPEG
                    </label>
                    <label className="flex items-center gap-2 text-sm font-medium cursor-pointer">
                      <input 
                        type="radio" value="png" 
                        checked={downloadFormat === 'png'} 
                        onChange={() => setDownloadFormat('png')} 
                        className="accent-indigo-600 w-4 h-4 cursor-pointer" 
                      />
                      PNG
                    </label>
                  </div>
                </div>

                {downloadFormat === 'jpeg' && (
                  <div className="animate-in fade-in slide-in-from-top-2 duration-200">
                    <label className="flex items-center justify-between text-sm font-semibold text-slate-700 mb-1.5">
                      <span>JPEG品質</span>
                      <span className="text-indigo-600 font-bold">{Math.round(jpegQuality * 100)}%</span>
                    </label>
                    <input 
                      type="range" min="0.1" max="1" step="0.1" value={jpegQuality} 
                      onChange={(e) => setJpegQuality(parseFloat(e.target.value))} 
                      className="w-full accent-indigo-600" 
                    />
                  </div>
                )}
              </div>
            </div>

            {/* ファイルリスト */}
            {images.length > 0 && (
              <div className="flex-1 flex flex-col min-h-[250px] max-h-[40vh] lg:max-h-none">
                <div className="flex items-center justify-between mb-3">
                  <h2 className="text-sm font-bold text-slate-800 uppercase tracking-widest flex items-center gap-2">
                    <FileImage size={16} className="text-slate-500"/> 選択済み ({images.length})
                  </h2>
                  <button onClick={clearAll} className="text-xs text-red-600 hover:text-red-700 font-bold px-3 py-1.5 rounded-lg hover:bg-red-50 transition-colors">
                    すべてクリア
                  </button>
                </div>
                <div className="flex-1 overflow-y-auto rounded-xl border border-slate-200 bg-slate-50 p-2 space-y-1.5 shadow-inner">
                  {images.map((img, idx) => (
                    <div key={idx} className="flex items-center justify-between bg-white p-2.5 rounded-lg shadow-sm border border-slate-100 text-sm group hover:border-slate-300 transition-colors">
                      <div className="flex items-center gap-2 overflow-hidden flex-1 pr-2">
                        <span className="text-slate-400 font-mono text-xs w-5 text-right">{idx + 1}.</span>
                        <span className="truncate text-slate-700 font-medium" title={img.name}>{img.name}</span>
                      </div>
                      <button onClick={() => removeImage(idx)} className="text-slate-300 hover:text-red-500 p-1.5 rounded-md hover:bg-red-50 transition-colors" title="削除">
                        <Trash2 size={16} />
                      </button>
                    </div>
                  ))}
                </div>
              </div>
            )}
          </div>
        </aside>

        {/* 右パネル: プレビュー */}
        <section className="flex-1 bg-slate-200/60 flex flex-col relative overflow-hidden">
          
          {/* エラーメッセージ */}
          {error && (
            <div className="absolute top-4 left-1/2 -translate-x-1/2 bg-red-50 border border-red-200 text-red-700 px-5 py-3 rounded-xl shadow-lg z-20 flex items-center gap-3 w-[90%] max-w-lg">
              <AlertCircle size={20} className="text-red-500 flex-shrink-0" />
              <span className="flex-1 text-sm font-medium">{error}</span>
              <button onClick={() => setError(null)} className="text-red-400 hover:text-red-700 font-bold p-1">&times;</button>
            </div>
          )}

          {/* アクションバー (ダウンロード等) */}
          <div className="p-4 flex justify-end gap-3 z-10 bg-white/70 backdrop-blur-md border-b border-slate-200 shadow-sm">
            <button 
              onClick={downloadImage}
              disabled={!previewUrl || isProcessing}
              className="flex items-center gap-2 bg-indigo-600 text-white px-6 py-2.5 rounded-lg font-bold shadow-md hover:bg-indigo-700 hover:shadow-lg focus:ring-4 focus:ring-indigo-100 disabled:bg-slate-300 disabled:text-slate-500 disabled:shadow-none disabled:cursor-not-allowed transition-all active:scale-95"
            >
              <Download size={18} />
              結合画像をダウンロード
            </button>
          </div>

          {/* プレビュー表示エリア */}
          <div className="flex-1 overflow-auto p-4 md:p-8 flex items-start justify-center relative scroll-smooth">
            
            {/* ローディング表示 */}
            {isProcessing && (
              <div className="absolute inset-0 bg-slate-100/50 backdrop-blur-[2px] flex items-center justify-center z-10 transition-opacity">
                 <div className="bg-white px-8 py-6 rounded-2xl shadow-xl flex flex-col items-center gap-4">
                   <div className="w-12 h-12 border-4 border-indigo-100 border-t-indigo-600 rounded-full animate-spin"></div>
                   <p className="text-indigo-900 font-bold text-sm tracking-wide">画像をレイアウト中...</p>
                 </div>
              </div>
            )}

            {/* 初期状態（空） */}
            {!previewUrl && !isProcessing && images.length === 0 && (
              <div className="text-slate-400 flex flex-col items-center justify-center h-full gap-5">
                <div className="w-32 h-32 rounded-full bg-slate-100 flex items-center justify-center border-4 border-white shadow-sm">
                  <ImageIcon size={48} className="text-slate-300" />
                </div>
                <p className="font-medium text-slate-500">左側のエリアに画像をドラッグ＆ドロップしてください</p>
              </div>
            )}

            {/* プレビュー画像 */}
            {previewUrl && (
              <div className="bg-white p-2 rounded-lg shadow-2xl ring-1 ring-slate-200/50 max-w-full">
                <img 
                  src={previewUrl} 
                  alt="結合プレビュー" 
                  className="max-w-full h-auto object-contain transition-opacity duration-300" 
                  style={{ opacity: isProcessing ? 0.3 : 1 }}
                />
              </div>
            )}
            
          </div>
        </section>
      </main>
    </div>
  );
}
